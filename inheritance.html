Inheritance Functionality in Javascript (Prototypal, Classical)

Until ECMAScript 5, there is no straightforward way to implement inheritance as in other programming languages. From ECMAScript 6, the paradigm has been changed with the introduction of classes, interfaces, extensions.

Here, you can see the simple example that describes the inheritance in javascript in EC5

		//Code
	function Animal(name) {
		this.name = name;
	}
	
	Animal.prototype.setName = function (name) {
		this.name = name;
	}
		
	Animal.prototype.getName = function () {
		return this.name + ':' + this.type;
	}
	
	function Cat (name) {
		Animal.prototype.setName.call(this, name);
		this.type = "Domestic";
	}
	Cat.prototype = new Animal;
	Cat.prototype.constructor = Animal;
	
	function Dog(name) {
		Animal.prototype.setName.call(this, name);
		this.type = 'Barking Domestic';
	}
	
	Dog.prototype = new Animal;
	Dog.prototype.constructor = Animal;
	
	
	
	var c = new Cat('Name of the cat');
	var d = new Dog('Dog Name');
	
	console.log(c.getName());
	console.log(d.getName());
